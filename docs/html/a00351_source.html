692  // c^{k+1, T} = c^{0, T} + \lambda^{k+1, T} (Ax-b) = c^{0, T} + \lambda^{k+1, T} A - \lambda^{k+1, T} b
693  // c^{k+1} = c^{0} + A^T \lambda^{k+1} - b^T \lambda^{k+1} = c^{k} + A^T \Delta \lambda^{k+1} - b^T \lambda^{k+1}
694  axpy(m_constrMatrix.numRows, (coefficient_value_type)-1, m_vLagMultiplier, m_vNewLagMultiplier); // m_vNewLagMultiplier becomes delta multipliers
695  ATxPlusy((coefficient_value_type)1, m_constrMatrix, m_vNewLagMultiplier, m_vObjCoef);
696  axpy(m_constrMatrix.numRows, (coefficient_value_type)1, m_vNewLagMultiplier, m_vLagMultiplier); // update delta multipliers of m_vNewLagMultiplier to m_vLagMultiplier
697 #else
698  updater->operator()(m_iter, m_constrMatrix.numRows, m_vSlackness, m_vLagMultiplier, m_vLagMultiplier);
699 
700  std::fill(m_vObjCoef, m_vObjCoef+m_model->numVariables(), 0);
701  for (typename std::vector<term_type>::const_iterator it = m_model->objective().terms().begin(), ite = m_model->objective().terms().end(); it != ite; ++it)
702  m_vObjCoef[it->variable().id()] += it->coefficient();
703  ATxPlusy((coefficient_value_type)1, m_constrMatrix, m_vLagMultiplier, m_vObjCoef);
704 #endif
705 
706  if (m_lagObjFlag)
707  {
708  m_objConstant = -dot(m_constrMatrix.numRows, m_vConstrRhs, m_vLagMultiplier);
709  }
710 }
711 template <typename T, typename V>
712 void MultiKnapsackLagRelax<T, V>::solveLag()
713 {
714  // for each item
715  CompareVariableByCoefficient helper (m_vObjCoef);
716  variable_value_type* vVariableSol = &m_model->variableSolutions()[0];
717  unsigned int i = 0;
718  variable_type const* variableGroupBegin = m_vGroupedVariable;
719  variable_type const* variableGroupEnd = m_vGroupedVariable;
720  variable_type const* variable;
721 
722  // reset variables in this group
723  std::fill(vVariableSol, vVariableSol+m_model->numVariables(), 0);
724  for (i = 0; i < m_numGroups; ++i)
725  {
726  variableGroupBegin = variableGroupEnd;
727  variableGroupEnd = m_vGroupedVariable+m_vVariableGroupBeginIndex[i+1];
728  // find the bin with minimum cost for each item
729  variable = std::min_element(variableGroupBegin, variableGroupEnd, helper);
730  vVariableSol[variable->id()] = 1;
731  }
732  // evaluate current objective
733  if (m_lagObjFlag)
734  m_lagObj = evaluateLagObjective();

752 template <typename T, typename V>
753 SolverProperty MultiKnapsackLagRelax<T, V>::converge()
754 {
755  bool feasibleFlag = true;
756  bool convergeFlag = true;
757  for (typename matrix_type::index_type i = 0; i < m_constrMatrix.numRows; ++i)
758  {
759  coefficient_value_type multiplier = m_vLagMultiplier[i];
760  coefficient_value_type slackness = m_vSlackness[i];
761  // must satisfy KKT condition
762  // lambda >= 0
763  // g(x) <= 0
764  // lambda * g(x) = 0
765  //
766  // g(x) = Ax-b = -slackness
767  if (slackness < 0)
768  {
769  convergeFlag = feasibleFlag = false;
770  break;
771  }
772  else if (multiplier*slackness != 0)
773  {
774  convergeFlag = false;
775  }
776  }
777  SolverProperty status = INFEASIBLE;
778  // store feasible solutions with better objective
779  if (feasibleFlag)
780  {
781  coefficient_value_type obj = m_model->evaluateObjective();
782  if (obj < m_bestObj)
783  {
784  m_vBestVariableSol = m_model->variableSolutions();
785  m_bestObj = obj;
786  }
787  status = SUBOPTIMAL;
788  }
789  if (convergeFlag)
790  status = OPTIMAL;
791 
792  return status;
793 }

797  if (status == OPTIMAL) // already OPTIMAL
