299 template <typename T, typename V>
300 void MinCostFlow<T, V>::buildGraph()
301 {
302  // compute how many nodes
303  // regard equality constraints as nodes except for s and t
304  unsigned int numNodes = 0;
305  for (typename std::vector<constraint_type>::const_iterator it = m_model->constraints().begin(), ite = m_model->constraints().end(); it != ite; ++it)
306  {
307  // assume only equality constraints
308  limboAssertMsg(it->sense() == '=', "only support equality constraints %s", m_model->constraintName(*it).c_str());
309  // equality constraints
310  numNodes += 1;
311  }
312 
313  // construct nodes
314  coefficient_value_type totalSupply = 0;
315  m_graph.reserveNode(numNodes+1); // additional nodes for s and t
316  for (unsigned int i = 0, ie = numNodes+1; i < ie; ++i)
317  m_graph.addNode();
318  // node st
319  node_type st = m_graph.nodeFromId(numNodes);
320 
321  // a map record arc to its two nodes
322  // arc id, source node id, target node id
323  // variable id, constraint id positive, constraint id negative
324  // I assume vVar2Constr sorts according to variable id
325  std::vector<std::pair<unsigned int, unsigned int> > vVar2Constr (m_model->numVariables(), std::make_pair(std::numeric_limits<unsigned int>::max(), std::numeric_limits<unsigned int>::max()));
326  unsigned int constrIndex = 0;
327  for (typename std::vector<constraint_type>::const_iterator it = m_model->constraints().begin(), ite = m_model->constraints().end(); it != ite; ++it)
328  {
329  constraint_type constr = *it;
330  // equality constraints
331  //if (constr.sense() == '=')
332  {
333  // whether need to inverse the sign of the constraint
334  // only need to check the first found term
335  for (typename std::vector<term_type>::const_iterator itt = constr.expression().terms().begin(), itte = constr.expression().terms().end(); itt != itte; ++itt)
336  {
337  term_type const& term = *itt;
338  std::pair<unsigned int, unsigned int> const& value = vVar2Constr[term.variable().id()];
339  if (term.coefficient() == 1)
340  {
341  // I want value.first is not set
342  if (value.first != std::numeric_limits<unsigned int>::max())
343  {
344  constr.scale(-1);
345  break;
346  }
347  else if (value.second != std::numeric_limits<unsigned int>::max())
348  {
349  break;
350  }
351  }
352  else if (term.coefficient() == -1)
353  {
354  // I want value.first is not set
355  if (value.second != std::numeric_limits<unsigned int>::max())
356  {
357  constr.scale(-1);
358  break;
359  }
360  else if (value.first != std::numeric_limits<unsigned int>::max())
361  {
362  break;
363  }
364  }
365  else
366  {
367  limboAssertMsg(0, "unsupported coefficient %g in constraint %d\n", double(term.coefficient()), constr.id());
368  }
369  }
370  // record map
371  for (typename std::vector<term_type>::const_iterator itt = constr.expression().terms().begin(), itte = constr.expression().terms().end(); itt != itte; ++itt)
372  {
373  term_type const& term = *itt;
374  std::pair<unsigned int, unsigned int>& value = vVar2Constr[term.variable().id()];
375  if (term.coefficient() == 1)
376  {
377  limboAssertMsg(value.first == std::numeric_limits<unsigned int>::max(),
378  "variable %s appears in more than 1 constraint with coefficient 1", m_model->variableName(term.variable()).c_str());
379  value.first = constrIndex;
380  }
381  else if (term.coefficient() == -1)
382  {
383  limboAssertMsg(value.second == std::numeric_limits<unsigned int>::max(),
384  "variable %s appears in more than 1 constraint with coefficient -1", m_model->variableName(term.variable()).c_str());
385  value.second = constrIndex;
386  }
387  }
388  // compute supply
389  // since here we know whether this constraint is inversed
390  m_mSupply[m_graph.nodeFromId(constrIndex)] = constr.rightHandSide();
391  totalSupply += constr.rightHandSide();
392  // next cosntraint
393  ++constrIndex;
394  }
395  }
396  // supply for s and t
397  m_mSupply[st] = -totalSupply;
398 
399  // construct arcs
400  m_graph.reserveArc(vVar2Constr.size());
401  // I assume vVar2Constr sorts according to variable id
402  unsigned int var = 0;
403  for (std::vector<std::pair<unsigned int, unsigned int> >::const_iterator it = vVar2Constr.begin(), ite = vVar2Constr.end(); it != ite; ++it, ++var)
404  {
405  unsigned int constrSrc = it->first;
406  unsigned int constrTgt = it->second;
407 
408  arc_type arc;
409  if (constrSrc == std::numeric_limits<unsigned int>::max()) // from s
410  {
411  arc = m_graph.addArc(st, m_graph.nodeFromId(constrTgt));
412  }
413  else if (constrTgt == std::numeric_limits<unsigned int>::max()) // to t
414  {
415  arc = m_graph.addArc(m_graph.nodeFromId(constrSrc), st);
416  }
417  else
418  {
419  arc = m_graph.addArc(m_graph.nodeFromId(constrSrc), m_graph.nodeFromId(constrTgt));
420  }
421  m_mLower[arc] = m_model->variableLowerBound(m_model->variable(var));
422  m_mUpper[arc] = m_model->variableUpperBound(m_model->variable(var));
423 #ifdef DEBUG_MINCOSTFLOW
424  limboAssert(arc == m_graph.arcFromId(var));
425 #endif
426  }

429  // everytime solver is called
