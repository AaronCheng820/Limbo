388 template <typename T, typename V>
389 void DualMinCostFlow<T, V>::prepare()
390 {
391  // big M should be larger than the summation of all non-negative supply in the graph
392  // because this is the largest possible amount of flows.
393  // The supply for each node is the coefficient of variable in the objective.
394  if (m_diffBigM == std::numeric_limits<value_type>::max()) // if not set
395  {
396  m_diffBigM = 0;
397  for (typename std::vector<term_type>::const_iterator it = m_model->objective().terms().begin(), ite = m_model->objective().terms().end(); it != ite; ++it)
398  {
399  if (it->coefficient() > 0)
400  m_diffBigM += it->coefficient();
401  }
402  }
403  // big M for bound constraints should be larger than that for differential constraints
404  // so the results are more controllable for INFEASIBLE models
405  if (m_boundBigM == std::numeric_limits<value_type>::max())
406  m_boundBigM = m_diffBigM<<1;
407 
408  // reset data members
409  m_reversedArcFlowCost = 0;

439 template <typename T, typename V>
440 unsigned int DualMinCostFlow<T, V>::mapDiffConstraint2Graph(bool countArcs)
441 {
442  // preparing arcs
443  // arcs constraints like xi - xj >= cij
444  // add arc from node i to node j with cost -cij and capacity unlimited
445 
446  unsigned int numArcs = m_model->constraints().size();
447  if (!countArcs) // skip in count arcs mode
448  {
449  // normalize to '>' format
450  for (typename std::vector<constraint_type>::iterator it = m_model->constraints().begin(), ite = m_model->constraints().end(); it != ite; ++it)
451  {
452  constraint_type& constr = *it;
453  limboAssertMsg(constr.expression().terms().size() == 2, "only support differential constraints like xi - xj >= cij");
454  constr.normalize('>');
455  }
456  for (typename std::vector<constraint_type>::const_iterator it = m_model->constraints().begin(), ite = m_model->constraints().end(); it != ite; ++it)
457  {
458  constraint_type const& constr = *it;
459  std::vector<term_type> const& vTerm = constr.expression().terms();
460  variable_type xi = (vTerm[0].coefficient() > 0)? vTerm[0].variable() : vTerm[1].variable();
461  variable_type xj = (vTerm[0].coefficient() > 0)? vTerm[1].variable() : vTerm[0].variable();
462 
463  addArcForDiffConstraint(m_graph.nodeFromId(xi.id()), m_graph.nodeFromId(xj.id()), constr.rightHandSide(), m_diffBigM);
464  }
465  }
466  return numArcs;
467 }
468 template <typename T, typename V>
469 unsigned int DualMinCostFlow<T, V>::mapBoundConstraint2Graph(bool countArcs)
470 {
471  // 3. arcs for variable bounds
472  // from node to additional node
473 
474  // check whether there is node with non-zero lower bound or non-infinity upper bound
475  unsigned int numArcs = 0;
476  for (unsigned int i = 0, ie = m_model->numVariables(); i < ie; ++i)
477  {
478  value_type lowerBound = m_model->variableLowerBound(variable_type(i));
479  value_type upperBound = m_model->variableUpperBound(variable_type(i));
480  if (lowerBound != limbo::lowest<value_type>())
481  ++numArcs;
482  if (upperBound != std::numeric_limits<value_type>::max())
483  ++numArcs;
484  }
485  if (!countArcs && numArcs) // skip in count arcs mode
486  {
487  // 3.1 create additional node
488  // its corresponding weight is the negative sum of weight for other nodes
489  node_type addlNode = m_graph.addNode();
490  value_type addlWeight = 0;
491  for (typename std::vector<term_type>::const_iterator it = m_model->objective().terms().begin(), ite = m_model->objective().terms().end(); it != ite; ++it)
492  addlWeight -= it->coefficient();
493  m_mSupply[addlNode] = addlWeight;
494 
495  // bound constraint is more important
496  // so it has higher cost than normal differential constraints
497  for (unsigned int i = 0, ie = m_model->numVariables(); i < ie; ++i)
498  {
499  value_type lowerBound = m_model->variableLowerBound(variable_type(i));
500  value_type upperBound = m_model->variableUpperBound(variable_type(i));
501  // has lower bound
502  // add arc from node to additional node with cost d and cap unlimited
503  if (lowerBound != limbo::lowest<value_type>())
504  addArcForDiffConstraint(m_graph.nodeFromId(i), addlNode, lowerBound, m_boundBigM);
505  // has upper bound
506  // add arc from additional node to node with cost u and capacity unlimited
507  if (upperBound != std::numeric_limits<value_type>::max())
508  addArcForDiffConstraint(addlNode, m_graph.nodeFromId(i), -upperBound, m_boundBigM);
509  }
510  }
511  return numArcs;
512 }
513 template <typename T, typename V>
514 void DualMinCostFlow<T, V>::addArcForDiffConstraint(typename DualMinCostFlow<T, V>::node_type xi, typename DualMinCostFlow<T, V>::node_type xj, typename DualMinCostFlow<T, V>::value_type cij, typename DualMinCostFlow<T, V>::value_type bigM)
515 {
516  // add constraint xi - xj >= cij
517  //
518  // negative arc cost can be fixed by arc reversal
519  // for an arc with i -> j, with supply bi and bj, cost cij, capacity uij
520  // reverse the arc to
521  // i <- j, with supply bi-uij and bj+uij, cost -cij, capacity uij
522 
523  if (cij <= 0)
524  {
525  arc_type arc = m_graph.addArc(xi, xj);
526  m_mCost[arc] = -cij;
527  //m_mLower[arc] = 0;
528  m_mUpper[arc] = bigM;
529  }
530  else // reverse arc
531  {
532  arc_type arc = m_graph.addArc(xj, xi); // arc reversal
533  m_mCost[arc] = cij;
534  //m_mLower[arc] = 0;
535  m_mUpper[arc] = bigM;
536  m_mSupply[xi] -= bigM;
537  m_mSupply[xj] += bigM;
538 
539  m_reversedArcFlowCost += cij*bigM;
540  }
541 }
